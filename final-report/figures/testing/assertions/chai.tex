			foo.should.have.length(3);

			expect(foo).to.equal('bar');
			expect(foo).to.have.length(3);
		\end{lstlisting}
        \caption{Expect style}
        \label{fig:expect_style}
    \end{subfigure}

		assert.equal(foo, 'bar');
		assert.lengthOf(foo, 3)
		\end{lstlisting}
        \caption{Assert style}
        \label{fig:assert_style}
    \end{subfigure}
